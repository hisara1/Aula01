import numpy as np
import matplotlib.pyplot as pl
x = np.linspace(0, 15, num = 2)
y = np.linspace(2, 17, num = 2)
z = np.linspace(0, 20, num = 2)
pl.plot(x)
pl.plot(y)

pl.title("Retas paralelas")
pl.show()
pl.title("Retas concorrentes")

pl.plot(y)
pl.plot(z)
pl.show()
